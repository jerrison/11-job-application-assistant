# Job Application Web UI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a high-performance web UI replacing the Textual TUI, with FastAPI backend, WebSocket real-time updates, and vanilla JS frontend.

**Architecture:** FastAPI serves REST API + WebSocket + static files. Single HTML page with hash routing. WebSocket pushes only diffs (changed jobs). Worker subprocess managed by the server. Inline editing with staged changes.

**Tech Stack:** Python 3.14, FastAPI, uvicorn, WebSocket, SQLite (WAL), vanilla JS, CSS (no framework)

**Spec:** `docs/superpowers/specs/2026-03-16-web-ui-design.md`

**Existing code:** FastAPI already in `pyproject.toml[web]`. Existing draft web at `scripts/draft_web.py` (port 8420). All DB functions in `scripts/job_db.py`. Worker management pattern in `scripts/job_tui.py`.

```python
if str(SCRIPT_DIR) not in sys.path:

from job_db import (

# ── Database ─────────────────────────────────────────────────────────────

def _get_conn() -> sqlite3.Connection:
    return init_db(DB_PATH, check_same_thread=False)

_conn: sqlite3.Connection | None = None

def get_conn() -> sqlite3.Connection:
    global _conn
    if _conn is None:
        _conn = _get_conn()
    return _conn


# ── Worker Management ────────────────────────────────────────────────────

_worker_proc: subprocess.Popen | None = None

def is_worker_running() -> bool:
    global _worker_proc
    if _worker_proc and _worker_proc.poll() is None:
        return True
    if WORKER_PID_FILE.exists():
        try:
            pid = int(WORKER_PID_FILE.read_text().strip())
            os.kill(pid, 0)
            return True
        except (ValueError, OSError):
            pass
    return False

def start_workers(num_workers: int = 3) -> None:
    global _worker_proc
    if is_worker_running():
        return
    worker_script = str(SCRIPT_DIR / "job_worker.py")
    cmd = ["uv", "run", "--project", str(PROJECT_ROOT), "python", worker_script,
           "--workers", str(num_workers), "--headless"]
    _worker_proc = subprocess.Popen(
        cmd, cwd=str(PROJECT_ROOT),
        stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL,
        start_new_session=True,
    )

def stop_workers() -> None:
    global _worker_proc
    if _worker_proc and _worker_proc.poll() is None:
        try:
            os.killpg(os.getpgid(_worker_proc.pid), signal.SIGTERM)
        except (OSError, ProcessLookupError):
            _worker_proc.terminate()
        try:
            _worker_proc.wait(timeout=10)
        except subprocess.TimeoutExpired:
            try:
                os.killpg(os.getpgid(_worker_proc.pid), signal.SIGKILL)
            except (OSError, ProcessLookupError):
                _worker_proc.kill()
        _worker_proc = None
    elif WORKER_PID_FILE.exists():
        try:
            pid = int(WORKER_PID_FILE.read_text().strip())
            os.killpg(os.getpgid(pid), signal.SIGTERM)
        except (ValueError, OSError, ProcessLookupError):
            pass
    # Also kill orphaned processes
    for pattern in ("job_worker", "apply.sh"):
        subprocess.run(["pkill", "-f", pattern], capture_output=True)
    # Reset stale jobs
    conn = get_conn()
    reset_stale_jobs(conn, stale_threshold_seconds=0)
    conn.execute(
        "UPDATE jobs SET status = 'queued', error_message = '', progress = '' "
        "WHERE status IN ('failed', 'fix_in_progress')"
    )
    conn.commit()


# ── WebSocket Manager ────────────────────────────────────────────────────

class ConnectionManager:
    def __init__(self):
        self.active: list[WebSocket] = []
        self._last_state: dict[int, str] = {}  # job_id -> updated_at

    async def connect(self, ws: WebSocket):
        await ws.accept()
        self.active.append(ws)

    def disconnect(self, ws: WebSocket):
        self.active.remove(ws)

    async def broadcast(self, message: dict):
        data = json.dumps(message, default=str)
        for ws in list(self.active):
            try:
                await ws.send_text(data)
            except Exception:
                self.active.remove(ws)

    def get_changed_jobs(self, conn: sqlite3.Connection) -> list[dict]:
        """Return jobs whose updated_at changed since last check."""
        jobs = query_jobs(conn, limit=500)
        changed = []
        new_state = {}
        for j in jobs:
            jid = j["id"]
            updated = j.get("updated_at", "")
            new_state[jid] = updated
            if self._last_state.get(jid) != updated:
                changed.append(j)
        # Detect deletions
        for jid in set(self._last_state) - set(new_state):
            changed.append({"id": jid, "_deleted": True})
        self._last_state = new_state
        return changed

manager = ConnectionManager()


# ── App Factory ──────────────────────────────────────────────────────────

def create_app() -> FastAPI:
    app = FastAPI(title="Job Applications")

    # Serve static files
    if STATIC_DIR.is_dir():
        app.mount("/static", StaticFiles(directory=str(STATIC_DIR)), name="static")

    @app.get("/api/health")
    def health():
        return {"status": "ok", "worker_running": is_worker_running()}

    return app


def main():
    import uvicorn
    import argparse
    parser = argparse.ArgumentParser(description="Job Application Web UI")
    parser.add_argument("--host", default="127.0.0.1", help="Bind host (use 0.0.0.0 for LAN)")
    parser.add_argument("--port", type=int, default=8420, help="Port (default: 8420)")
    parser.add_argument("--workers", type=int, default=3, help="Number of job workers")
    args = parser.parse_args()

    app = create_app()
    start_workers(args.workers)
    try:
        uvicorn.run(app, host=args.host, port=args.port)
    finally:
        stop_workers()


if __name__ == "__main__":
    main()
```

```python
    # ── Request models ───────────────────────────────────────────────
    class AddJobsRequest(BaseModel):
        urls: list[str]
        provider: str | None = None
        priority: int = 0

    class BoardUrlRequest(BaseModel):
        url: str

    class DraftOverridesRequest(BaseModel):
        overrides: dict[str, Any]

    # ── Job endpoints ────────────────────────────────────────────────
    @app.get("/api/jobs")
    def list_jobs(status: str | None = None, board: str | None = None,
                  search: str | None = None, limit: int = 200, offset: int = 0):
        conn = get_conn()
        return query_jobs(conn, status=status, board=board, search=search,
                         limit=limit, offset=offset)

    @app.get("/api/jobs/{job_id}")
    def get_job_detail(job_id: int):
        conn = get_conn()
        job = get_job(conn, job_id)
        if not job:
            raise HTTPException(404, "Job not found")
        job["timeline"] = get_job_timeline(conn, job_id)
        job["metrics"] = get_job_metrics(conn, job_id)
        job["field_corrections"] = get_field_corrections(conn, job_id)
        return job

    @app.post("/api/jobs")
    def add_jobs(req: AddJobsRequest):
        conn = get_conn()
        added = 0
        duplicates = 0
        for url in req.urls:
            url = url.strip()
            if not url or not url.startswith("http"):
                continue
            try:
                add_job(conn, url, priority=req.priority, provider=req.provider)
                added += 1
            except sqlite3.IntegrityError:
                duplicates += 1
        return {"added": added, "duplicates": duplicates}

    @app.post("/api/jobs/{job_id}/approve")
    def approve(job_id: int):
        conn = get_conn()
        ok = approve_job(conn, job_id)
        if not ok:
            raise HTTPException(400, "Cannot approve — not in draft status")
        ensure_job_metrics(conn, job_id)
        m = get_job_metrics(conn, job_id)
        if m:
            update_job_metrics(conn, job_id, manual_interventions=m["manual_interventions"] + 1)
        return {"status": "approved"}

    @app.post("/api/jobs/{job_id}/reject")
    def reject(job_id: int):
        conn = get_conn()
        update_status(conn, job_id, "needs_manual", error_message="Rejected via web UI")
        return {"status": "rejected"}

    @app.post("/api/jobs/{job_id}/regenerate")
    def regenerate(job_id: int):
        conn = get_conn()
        ok = regenerate_job(conn, job_id)
        if not ok:
            raise HTTPException(400, "Cannot regenerate — not in draft status")
        return {"status": "regenerating"}

    @app.post("/api/jobs/{job_id}/retry")
    def retry(job_id: int):
        conn = get_conn()
        job = get_job(conn, job_id)
        if not job or job["status"] not in ("failed", "skipped_captcha", "skipped_auth",
                                             "needs_manual", "needs_board_url"):
            raise HTTPException(400, "Cannot retry this job")
        update_status(conn, job_id, "queued", error_message="", progress="")
        return {"status": "queued"}

    @app.post("/api/jobs/{job_id}/skip")
    def skip(job_id: int):
        conn = get_conn()
        update_status(conn, job_id, "skipped_captcha", error_message="Skipped via web UI")
        return {"status": "skipped"}

    @app.post("/api/jobs/{job_id}/board-url")
    def set_board_url(job_id: int, req: BoardUrlRequest):
        conn = get_conn()
        update_status(conn, job_id, "queued", board_url=req.url, canonical_url=req.url,
                      error_message="")
        log_event(conn, job_id, "board_url_set_manually", detail=req.url)
        return {"status": "queued"}

    @app.post("/api/jobs/{job_id}/draft-overrides")
    def save_draft_overrides(job_id: int, req: DraftOverridesRequest):
        conn = get_conn()
        job = get_job(conn, job_id)
        if not job or not job.get("output_dir"):
            raise HTTPException(400, "No output directory")
        overrides_path = Path(job["output_dir"]) / "draft_overrides.json"
        overrides_path.write_text(json.dumps(req.overrides, indent=2), encoding="utf-8")
        return {"status": "saved", "fields": len(req.overrides)}

    @app.delete("/api/jobs/{job_id}")
    def delete_job(job_id: int):
        conn = get_conn()
        conn.execute("DELETE FROM events WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM fix_attempts WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM provider_runs WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM job_phase_durations WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM field_corrections WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM job_metrics WHERE job_id = ?", (job_id,))
        conn.execute("DELETE FROM jobs WHERE id = ?", (job_id,))
        conn.commit()
        return {"status": "deleted"}
```

```python
    @app.websocket("/ws")
    async def websocket_endpoint(ws: WebSocket):
        await manager.connect(ws)
        try:
            # Send initial bulk
            conn = get_conn()
            jobs = query_jobs(conn, limit=500)
            await ws.send_json({"type": "job_bulk", "jobs": jobs})
            # Also send worker status
            await ws.send_json({"type": "worker_status", "running": is_worker_running()})
            # Keep alive — client can send pings
            while True:
                await ws.receive_text()
        except WebSocketDisconnect:
            manager.disconnect(ws)

    @app.on_event("startup")
    async def start_broadcast_loop():
        async def _broadcast_changes():
            while True:
                await asyncio.sleep(2)
                conn = get_conn()
                changed = manager.get_changed_jobs(conn)
                if changed:
                    for job in changed:
                        if job.get("_deleted"):
                            await manager.broadcast({"type": "job_deleted", "id": job["id"]})
                        else:
                            await manager.broadcast({"type": "job_update", "job": job})
                # Also broadcast worker status periodically
                await manager.broadcast({"type": "worker_status", "running": is_worker_running()})
        asyncio.create_task(_broadcast_changes())
```

```python
    finally:

    return 0
```

```python
    @app.get("/api/jobs/{job_id}/content/{filename}")
    def get_job_content(job_id: int, filename: str):
        """Read a content file from the job's output directory."""
        conn = get_conn()
        job = get_job(conn, job_id)
        if not job or not job.get("output_dir"):
            raise HTTPException(404, "No output directory")
        # Whitelist allowed filenames
        allowed = {"resume_content.json", "cover_letter_text.txt",
                   "jd_parsed.json", "draft_overrides.json"}
        if filename not in allowed:
            # Check for autofill report pattern
            if not filename.endswith("_autofill_report.json"):
                raise HTTPException(403, "File not allowed")
        path = Path(job["output_dir"])
        # Search in content/ and submit/ subdirs
        for subdir in ("content", "submit", ""):
            candidate = path / subdir / filename if subdir else path / filename
            if candidate.exists():
                text = candidate.read_text(encoding="utf-8")
                if filename.endswith(".json"):
                    return json.loads(text)
                return {"text": text}
        raise HTTPException(404, f"{filename} not found")
```
